## 3) รันกับคอลัมน์ answer และ “บังคับ schema” ตอน chat()
##    หมายเหตุ: ตั้ง .temperature = 0 ให้เสถียร
label_llm <- function(data, text_col = answer, model = "gpt-5-nano"){

text_sym <- rlang::enquo(text_col)
txt_vec <- data |> pull(!!text_sym) |> as.character()

message <- tibble(
    role = c("system","user"),
    content = c(system_prompt, glue::glue(user_prompt, TEXT = txt_vec))
)

out <- message |> 
    df_llm_message() |> 
    chat(openai(),
    .model = model,
    .json_schema = mainidea_schema) |> 
    get_reply_data()

return(out)
}

ทดลองใช้งานฟังก์ชันกับข้อมูลจริง
